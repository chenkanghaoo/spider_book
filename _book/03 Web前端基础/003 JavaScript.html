示例

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>鸣人学院</title>

    <style>
        *{
            padding: 0;
            margin: auto;
        }

        p{
            color: red;
        }

        body{
            background-color: green;
        }
        .btn{
            background-color: blue;
            color: white;
        }
    </style>
</head>
<body>
你好
<p id="content">欢迎来到鸣人学院</p>

<form action="">
    用户名: <input type="text" name="" id="">
    密码: <input type="password" name="" id="">
    <button class="btn">登录</button>
</form>

<script>
    alert("hello world");
    var x = 1;
    var y =2;
    console.log(x+y);

    if (x+y===2){
        x = document.getElementById("content");
        x.innerHTML="欢迎来到我家"
    }

</script>
</body>
</html>
